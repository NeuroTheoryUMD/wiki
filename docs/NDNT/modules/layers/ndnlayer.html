194        """
195        Initialize the weights and bias parameters of the layer.
196
197        Args:
198            weights_initializer: str, type of weight initialization to use
199                options: 'xavier_uniform', 'xavier_normal', 'kaiming_uniform', 'kaiming_normal', 'orthogonal', 'zeros', 'ones'
200            bias_initializer: str, type of bias initialization to use
201                options: 'uniform', 'normal', 'zeros', 'ones'
202            param: float or list of floats, parameter for weight initialization
203        """
204        # Default initializer, although others possible
205        if weights_initializer is None:
206            weights_initializer = 'uniform'
207        if bias_initializer is None:
208            bias_initializer = 'zeros'
209
210        if weights_initializer == 'uniform':
211            if param is None:
212                #param = 5**0.5
213                param = 1.0/self.num_filters
214            init.kaiming_uniform_(self.weight, a=param)
215        elif weights_initializer == 'zeros':
216            init.zeros_(self.weight)
217        elif weights_initializer == 'ones':
218            init.ones_(self.weight)
219        elif weights_initializer == 'normal':
220            if param is None:
221                mean = 0.0

223            elif isinstance(param, list):
224                mean, std = param

226                RuntimeError("if using 'normal', param must be None or a list of length 2")
227            init.normal_(self.weight, mean=mean, std=std)
228
229        elif weights_initializer == 'xavier_uniform':  # also known as Glorot initialization
230            if param is None:
231                param = 1.0
232            init.xavier_uniform_(self.weight, gain=param)  # set gain based on weights?
233        elif weights_initializer == 'xavier_normal':  # also known as Glorot initialization
234            if param is None:
235                param = 1.0
236            init.xavier_normal_(self.weight, gain=param)  # set gain based on weights?
237        else:
238            print('weights initializer not defined')

240        if self.pos_constraint>0:
241            self.weight.data = torch.sqrt(abs(self.weight.data))
242        # Pos-constraints will be implemented in preprocessing (including making negative there)

343    def get_weights(self, to_reshape=True, time_reverse=False, num_inh=0):
344        """
345        num-inh can take into account previous layer inhibition weights.
346        
347        Args:
348            to_reshape: bool, whether to reshape the weights to the original filter shape
349            time_reverse: bool, whether to reverse the time dimension
350            num_inh: int, number of inhibitory units
351
352        Returns:
353            ws: np.ndarray, weights of the layer, on the CPU
354        """
355
356        ws = self.preprocess_weights().detach().cpu().numpy()
357        num_filts = ws.shape[-1]
358        if time_reverse or (num_inh>0):
359            ws_tmp = np.reshape(ws, self.filter_dims + [num_filts])
360            num_lags = self.filter_dims[3]
361            if time_reverse and (num_lags > 1):
362                ws_tmp = ws_tmp[:, :, :, range(num_lags-1, -1, -1), :] # time reversal here
363            if num_inh > 0:
364                ws_tmp[range(-num_inh, 0), ...] *= -1
365            ws = ws_tmp.reshape((-1, num_filts))
366        if to_reshape:
367            ws = ws.reshape(self.filter_dims + [num_filts]).squeeze()
368        else:
369            ws = ws.squeeze()
370
371        if num_filts == 1:
372            # Add singleton dimension corresponding to number of filters
373            ws = ws[..., None]
374        return ws

345    def get_weights(self, to_reshape=True, time_reverse=False, num_inh=0):
346        """
347        num-inh can take into account previous layer inhibition weights.
348        
349        Args:
350            to_reshape: bool, whether to reshape the weights to the original filter shape
351            time_reverse: bool, whether to reverse the time dimension
352            num_inh: int, number of inhibitory units
353
354        Returns:
355            ws: np.ndarray, weights of the layer, on the CPU
356        """
357
358        ws = self.preprocess_weights().detach().cpu().numpy()
359        num_filts = ws.shape[-1]
360        if time_reverse or (num_inh>0):
361            ws_tmp = np.reshape(ws, self.filter_dims + [num_filts])
362            num_lags = self.filter_dims[3]
363            if time_reverse and (num_lags > 1):
364                ws_tmp = ws_tmp[:, :, :, range(num_lags-1, -1, -1), :] # time reversal here
365            if num_inh > 0:
366                ws_tmp[range(-num_inh, 0), ...] *= -1
367            ws = ws_tmp.reshape((-1, num_filts))
368        if to_reshape:
369            ws = ws.reshape(self.filter_dims + [num_filts]).squeeze()
370        else:
371            ws = ws.squeeze()
372
373        if num_filts == 1:
374            # Add singleton dimension corresponding to number of filters
375            ws = ws[..., None]
376        return ws
