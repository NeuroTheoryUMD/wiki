113        """
114        Forward pass through the Tlayer.
115
116        Args:
117            x: torch.Tensor, input tensor
118
119        Returns:
120            y: torch.Tensor, output tensor
121        """
122        # Reshape stim matrix LACKING temporal dimension [bcwh] 
123        s = (x.T)[None, :, :] # [B,dims]->[1,dims,B]
124
125        w = self.preprocess_weights()
126        w = w.reshape([self.folded_dims, self.filter_dims[3], -1]).permute(2,0,1) # [C,T,N]->[N,C,T]
127
128        # pad the batch dimension
129        pad = (self.filter_dims[-1]-1, 0)
130        s = F.pad(s, pad, "constant", 0)
131
132        y = F.conv1d(
133            s,
134            w, 
135            bias=self.bias,
136            stride=1, dilation=1)
137        
138        y = y.permute(2,1,0)[:, :, 0] # [1,N,B] -> [B,N,1] -> [B, N]
139    
140        if self.output_norm is not None:
141            y = self.output_norm(y)
142
143        # Nonlinearity
144        if self.NL is not None:
145            y = self.NL(y)
146        
147        if self._ei_mask is not None:
148            y = y * self._ei_mask[None,:]
149        
150        if self.res_layer:
151            y = y + x
152        #y = y.reshape((-1, self.num_outputs))
153
154        # store activity regularization to add to loss later
155        #self.activity_regularization = self.activity_reg.regularize(y)
156        if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model
157            self.reg.compute_activity_regularization(y)
