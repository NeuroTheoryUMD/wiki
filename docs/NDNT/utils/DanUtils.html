 68def find_peaks( x, clearance=10, max_peaks=10, thresh=13.0 ):
 69    """Find maximum of peaks and then get rid of other points around it for plus/minus some amount"""
 70    y = deepcopy(x)
 71    rem = np.arange(len(x))
 72    pks, amps = [], []
 73    for counter in range(max_peaks):
 74        #plt.plot(y,clrs[counter])
 75        a = rem[np.argmax(y[rem])]
 76        if y[a] >= thresh:
 77            pks.append(a)
 78            amps.append(y[a])
 79        rem = np.setdiff1d(rem, np.arange(a-clearance, a+clearance+1), assume_unique=True )
 80        
 81    pks = np.array(pks, dtype=np.int64)
 82    amps = np.array(amps, dtype=np.float32)
 83
 84    return pks, amps
 85# END find_peaks

 88def filename_num2str( n, num_digits=2 ):
 89    if n == 0:

162def fold_sample( num_items, folds=5, random_gen=False, which_fold=None):
163    """Divide fold sample deterministically or randomly distributed over number of items. More options
164    can be added, but his captures the basics."""
165    if random_gen:
166        num_val = int(num_items/folds)
167        tmp_seq = np.random.permutation(num_items)
168        val_items = np.sort(tmp_seq[:num_val])
169        rem_items = np.sort(tmp_seq[num_val:])
170    else:
171        if which_fold is None:
172            offset = int(folds//2) # sample middle folds as test_data
173        else:
174            assert which_fold < folds
175            offset = which_fold
176        val_items = np.arange(offset, num_items, folds, dtype='int64')
177        rem_items = np.delete(np.arange(num_items, dtype='int64'), val_items)
178    return val_items, rem_items

182def time_embedding_simple( stim, nlags ):
183    """Simple time embedding: takes the stim and time-embeds with nlags
184    If stim is multi-dimensional, it flattens. This is a numpy function"""
185
186    if len(stim.shape) == 1:
187        stim = stim[:, None]
188    NT, num_dims = stim.shape
189    tmp_stim = deepcopy(stim).reshape([NT, -1])  # automatically generates 2-dimensional stim
190
191    X = np.zeros([NT, num_dims, nlags], dtype=np.float32)
192    X[:, :, 0] = tmp_stim
193    for ii in range(1, nlags):
194        X[ii:, :, ii] = tmp_stim[:(-ii),:]
195    return X.reshape([NT, -1])
196# END time_embedding_simple()

199def design_matrix_drift( NT, anchors, zero_left=True, zero_right=False, const_right=False, to_plot=False):
200    """Produce a design matrix based on continuous data (s) and anchor points for a tent_basis.
201    Here s is a continuous variable (e.g., a stimulus) that is function of time -- single dimension --
202    and this will generate apply a tent basis set to s with a basis variable for each anchor point. 
203    The end anchor points will be one-sided, but these can be dropped by changing "zero_left" and/or
204    "zero_right" into "True".
205
206    Inputs: 
207        NT: length of design matrix
208        anchors: list or array of anchor points for tent-basis set
209        zero_left, zero_right: boolean whether to drop the edge bases (default for both is False)
210    Outputs:
211        X: design matrix that will be NT x the number of anchors left after zeroing out left and right
212    """
213    anchors = list(anchors)
214    if anchors[0] > 0:
215        anchors = [0] + anchors
216    #if anchors[-1] < NT:
217    #    anchors = anchors + [NT]
218    NA = len(anchors)
219
220    X = np.zeros([NT, NA])
221    for aa in range(NA):
222        if aa > 0:
223            dx = anchors[aa]-anchors[aa-1]
224            X[range(anchors[aa-1], anchors[aa]), aa] = np.arange(dx)/dx
225        if aa < NA-1:
226            dx = anchors[aa+1]-anchors[aa]
227            X[range(anchors[aa], anchors[aa+1]), aa] = 1-np.arange(dx)/dx
228
229    if zero_left:
230        X = X[:, 1:]
231
232    if const_right:
233        X[range(anchors[-1], NT), -1] = 1.0
234
235    if zero_right:
236        X = X[:, :-1]
237
238    if to_plot:
239        plt.imshow(X.T, aspect='auto', interpolation='none')
240        plt.show()
241
242    return X

245def max_multiD(k):
246    num_dims = len(k.shape)
247    if num_dims == 2:
248        a,b = k.shape
249        d1best = np.argmax( np.max(k, axis=0) )
250        d0best = np.argmax( k[:, d1best] )
251        return [d0best, d1best] 
252    elif num_dims == 3:
253        a,b,c = k.shape
254        d2best = np.argmax( np.max(np.reshape(k, [a*b, c]), axis=0) )
255        d1best = np.argmax( np.max(k[:,:, d2best], axis=0) )
256        d0best = np.argmax( k[:,d1best, d2best] )
257        return [d0best, d1best, d2best]
258    elif num_dims == 4:
259        a,b,c,d = k.shape
260        d3best = np.argmax( np.max(np.reshape(k, [a*b*c, d]), axis=0) )
261        d2best = np.argmax( np.max(np.reshape(k[..., d3best], [a*b, c]), axis=0) )
262        d1best = np.argmax( np.max(k[:,:, d2best, d3best], axis=0) )
263        d0best = np.argmax( k[:,d1best, d2best, d3best] )
264        return [d0best, d1best, d2best, d3best]
265    else:
266        print('havent figured out how to do this number of dimensions yet.')
267# END design_matrix_drift

270def iterate_lbfgs(mod, dat, lbfgs_pars, train_inds=None, val_inds=None, 
271                  tol=0.0001, max_iter = 20, verbose=True ):
272    
273    if train_inds is None:
274        train_inds = dat.train_inds
275    if val_inds is None:
276        val_inds = dat.val_inds
277
278    cont = True
279    iter = 0
280    while cont and (iter < max_iter):
281        LLprev = mod.eval_models(dat[val_inds], null_adjusted=False)[0]
282        save_mod = deepcopy(mod)
283        mod.fit(dat, train_inds=train_inds, val_inds=val_inds,
284                force_dict_training=True, **lbfgs_pars, version=9, verbose=0, seed=iter)
285        LLcur = mod.eval_models(dat[val_inds], null_adjusted=False)[0]
286        if LLprev-LLcur < tol:
287            cont = False
288            if LLprev < LLcur:
289                mod = save_mod
290                LLcur = LLprev
291                if verbose:
292                    print("    Reverting")
293            else:
294                if verbose:
295                    print("    Below tolerance", LLprev-LLcur)
296        else:
297            if verbose:
298                print( "  Iter %d: %0.6f"%(iter, LLcur) )
299        iter += 1
300    return mod, LLcur

303def binocular_data_import( datadir, expt_num ):
304    """Usage: stim, Robs, DFs, used_inds, Eadd_info = binocular_data_import( datadir, expt_num )
305
306    Inputs:
307        datadir: directory on local drive where datafiles are
308        expt_num: the experiment number (1-12) representing the binocular experiments we currently have. All
309                    datafiles are called 'BS2expt?.mat'. Note that numbered from 1 (not zero)
310    
311    Outputs:
312        stim: formatted as NT x 72 (stimuli in each eye are cropped to NX=36). It will be time-shifted by 1 to 
313                eliminate 0-latency stim
314        Robs: response concatenating all SUs and MUs, NT x NC. NSU is saved as part of Eadd_info
315        DFs:  data_filters for experiment, also NT x NC (note MU datafilters are initialized to 1)
316        used_inds: indices overwhich data is valid according to initial data parsing (adjusted to python) 
317        Eadd_info: dictionary containing all other relevant info for experiment
318    """
319
320    # Constants that are pretty much set for our datasets
321    stim_trim = np.concatenate( (range(3,39), range(45,81)))
322    time_shift = 1
323    NX = 36
324
325    # Read all data into memory
326    filename = 'B2Sexpt'+ str(expt_num) + '.mat'
327    Bmatdat = sio.loadmat(datadir+filename)
328    stim = NDNutils.shift_mat_zpad(Bmatdat['stim'][:,stim_trim], time_shift, 0)
329
330    #NX = int(stim.shape[1]) // 2
331    RobsSU = Bmatdat['RobsSU']
332    RobsMU = Bmatdat['RobsMU']
333    #MUA = Bmatdata[nn]['RobsMU']
334    #SUA = Bmatdata[nn]['RobsSU']
335    NTtot, numSUs = RobsSU.shape
336
337    data_filtersSU = Bmatdat['SUdata_filter']
338    data_filtersMU = np.ones( RobsMU.shape, dtype='float32')  # note currently no MUdata filter but their needs to be....
339
340    Robs = np.concatenate( (RobsSU, RobsMU), axis=1 )
341    DFs = np.concatenate( (data_filtersSU, data_filtersMU), axis=1 )
342
343    # Valid and cross-validation indices
344    used_inds = np.add(np.transpose(Bmatdat['used_inds'])[0,:], -1) # note adjustment for python v matlab indexing
345    Ui_analog = Bmatdat['Ui_analog'][:,0]  # these are automaticall in register
346    XiA_analog = Bmatdat['XiA_analog'][:,0]
347    XiB_analog = Bmatdat['XiB_analog'][:,0]
348    # two cross-validation datasets -- for now combine
349    Xi_analog = XiA_analog+XiB_analog  # since they are non-overlapping, will make 1 in both places
350
351    # Derive full-dataset Ui and Xi from analog values
352    Ui = np.intersect1d(used_inds, np.where(Ui_analog > 0)[0])
353    Xi = np.intersect1d(used_inds, np.where(Xi_analog > 0)[0])
354
355    NC = Robs.shape[1]
356    NT = len(used_inds)
357
358    dispt_raw = NDNutils.shift_mat_zpad( Bmatdat['all_disps'][:,0], time_shift, 0 ) # time shift to keep consistent
359    # this has the actual disparity values, which are at the resolution of single bars, and centered around the neurons
360    # disparity (sometime shifted to drive neurons well)
361    # Sometimes a slightly disparity is used, so it helps to round the values at some resolution
362    dispt = np.round(dispt_raw*100)/100
363    frs = NDNutils.shift_mat_zpad( Bmatdat['all_frs'][:,0], time_shift, 0 )
364    corrt = NDNutils.shift_mat_zpad( Bmatdat['all_corrs'][:,0], time_shift, 0 )
365    # Make dispt consistent with corrt (early experiments had dispt labeled incorrectly)
366    corr_funny = np.where((corrt == 0) & (dispt != -1005))[0]
367    if len(corr_funny) > 0:
368        print( "Warning: %d indices have corr=0 but labeled disparity."%len(corr_funny) )
369        dispt[corr_funny] = -1005
370
371    disp_list = np.unique(dispt)
372    # where it is -1009 this corresponds to a blank frame
373    # where it is -1005 this corresponds to uncorrelated images between the eyes
374
375    if Bmatdat['rep_inds'] is None:
376        #rep_inds = [None]*numSUs
377        rep_inds = None
378    elif len(Bmatdat['rep_inds'][0][0]) < 10:
379        rep_inds = None
380    else:
381        rep_inds = []
382        for cc in range(numSUs):
383            rep_inds.append( np.add(Bmatdat['rep_inds'][0][cc], -1) ) 
384
385    print( "Expt %d: %d SUs, %d total units, %d out of %d time points used."%(expt_num, numSUs, NC, NT, NTtot))
386    #print(len(disp_list), 'different disparities:', disp_list)
387
388    Eadd_info = {
389        'Ui_analog': Ui_analog, 'XiA_analog': XiA_analog, 'XiB_analog': XiB_analog, 'Xi_analog': Xi_analog,
390        'Ui': Ui, 'Xi': Xi, # these are for use with data_filters and inclusion of whole experiment,
391        'NSU': numSUs, 'NC': NC,
392        'dispt': dispt, 'corrt': corrt, 'disp_list': disp_list, 
393        'frs': frs, 'rep_inds': rep_inds}

439
440    # Add some things to Einfo
441    Einfo['used_inds_all'] = used_inds
442    Einfo['XiA_cell'] = XVi1
443    Einfo['XiB_cell'] = XVi2
444    Einfo['Robs'] = Robs # all cells, full indexing
445
446    print( "Adjusted for cell %d: %d time points"%(cell_num, NT))
447
448    return stim_all, Robs, used_cell, Ui, Xi, Einfo

451def monocular_data_import( datadir, exptn, time_shift=1, num_lags=20):
452    """
453    Usage: stim, Robs, DFs, used_inds, Eadd_info = binocular_data_import( datadir, expt_num )
454    Note: expt num is starting with 1
455    block_output determines whether to fit using block info, or used_inds info (derived from blocks)
456    num_lags is for purposes of used_inds
457    """
458    import scipy.io as sio
459
460    l_files_to_use = np.add(list(range(16)), 1)  # good laminar-probe experiments
461    u_files_to_use = [1, 2, 5, 6, 12]  # good utah-array experiments
462    assert exptn <= len(l_files_to_use)+len(u_files_to_use), 'Expt number too high.'
463    if exptn <= len(l_files_to_use):
464        filename = 'expt'
465        ee = l_files_to_use[exptn-1]
466        is_utah = False
467    else:
468        filename = 'Uexpt'
469        #utah_array[nn] = True
470        ee = u_files_to_use[exptn-1 - len(l_files_to_use)]
471        is_utah = True
472    if ee < 10:
473        filename += '0'
474    filename += str(ee) + '.mat'         
475    matdata = sio.loadmat( datadir+filename )
476
477    sus = np.squeeze(matdata['goodSUs']) - 1  # switch from matlab indexing
478    mus = np.squeeze(matdata['goodMUs']) - 1
479
480    #print('SUs:', sus)
481    NC = len(sus)
482    layers = matdata['layers'][0,:]
483    block_list = matdata['block_inds'] # note matlab indexing
484    stim_all = NDNutils.shift_mat_zpad(matdata['stimulus'], time_shift, 0)
485    NTtot, NX = stim_all.shape
486    DFs = deepcopy(matdata['data_filters'][:,sus])
487    Robs = deepcopy(matdata['binned_SU'][:,sus])
488    RobsMU = deepcopy(matdata['binned_MUA'][:, mus])
489    DFsMU = deepcopy(matdata['data_filters_MUA'][:,mus])
490
491    # Break up into train and test blocks
492    # Assemble train and test indices based on BIlist
493    NBL = block_list.shape[0]
494    Xb = np.arange(2, NBL, 5)  # Every fifth trial is cross-validation
495    Ub = np.array(list(set(list(range(NBL)))-set(Xb)), dtype='int')
496    
497    # Make block indxs #used_inds = make_block_inds( block_list, gap=num_lags )
498    # Also further modify DFs to incorporate gaps (for this purpose) 
499    used_inds = []
500    for nn in range(block_list.shape[0]):
501        used_inds = np.concatenate( 
502            (used_inds, 
503            np.arange(block_list[nn,0]-1+num_lags, block_list[nn,1], dtype='int')),
504            axis=0)
505        DFs[np.arange(block_list[nn,0]-1, block_list[nn,0]+num_lags, dtype='int'), :] = 0.0

200def design_matrix_drift( NT, anchors, zero_left=True, zero_right=False, const_right=False, to_plot=False):
201    """Produce a design matrix based on continuous data (s) and anchor points for a tent_basis.
202    Here s is a continuous variable (e.g., a stimulus) that is function of time -- single dimension --
203    and this will generate apply a tent basis set to s with a basis variable for each anchor point. 
204    The end anchor points will be one-sided, but these can be dropped by changing "zero_left" and/or
205    "zero_right" into "True".
206
207    Inputs: 
208        NT: length of design matrix
209        anchors: list or array of anchor points for tent-basis set
210        zero_left, zero_right: boolean whether to drop the edge bases (default for both is False)
211    Outputs:
212        X: design matrix that will be NT x the number of anchors left after zeroing out left and right
213    """
214    anchors = list(anchors)
215    if anchors[0] > 0:
216        anchors = [0] + anchors
217    #if anchors[-1] < NT:
218    #    anchors = anchors + [NT]
219    NA = len(anchors)
220
221    X = np.zeros([NT, NA])
222    for aa in range(NA):
223        if aa > 0:
224            dx = anchors[aa]-anchors[aa-1]
225            X[range(anchors[aa-1], anchors[aa]), aa] = np.arange(dx)/dx
226        if aa < NA-1:
227            dx = anchors[aa+1]-anchors[aa]
228            X[range(anchors[aa], anchors[aa+1]), aa] = 1-np.arange(dx)/dx
229
230    if zero_left:
231        X = X[:, 1:]
232
233    if const_right:
234        X[range(anchors[-1], NT), -1] = 1.0
235
236    if zero_right:
237        X = X[:, :-1]
238
239    if to_plot:
240        plt.imshow(X.T, aspect='auto', interpolation='none')
241        plt.show()
242
243    return X

Outputs:

stim_all: formatted as NT x 72 (stimuli in each eye are cropped to NX=36). It will be time-shifted by 1 to eliminate 0-latency stim. Note this is all stim in experiment, val_inds from used_inds... Robs: response concatenating all SUs and MUs, NT x NC. NSU and full Robs is saved as part of Eadd_info. This is already selected by used_inds_cell, so no need for further reduction used_inds_cell: indices overwhich data is valid for that particular cell. Should be applied to stim only UiC, XiC: cross-validation indices for that cell, based on used_inds_cell Eadd_info: dictionary containing all other relevant info for experiment

if the full experiment is NT long, stim and Robs are also NT long, stim[used_inds_cell,:] and Robs[used_inds_cell] are the valid data for that cell, and UiC and XiC are the proper train and test indices of used_inds_cell to access, i.e. _model.train( input_data=stim[used_inds_cell,:], output_data=Robs[used_inds_cell], train_indxs=UiC, ...)
