  20def polyinterp(points, x_min_bound=None, x_max_bound=None, plot=False):
  21    """
  22    Gives the minimizer and minimum of the interpolating polynomial over given points
  23    based on function and derivative information. Defaults to bisection if no critical
  24    points are valid.
  25    Based on polyinterp.m Matlab function in minFunc by Mark Schmidt with some slight
  26    modifications.
  27    Implemented by: Hao-Jun Michael Shi and Dheevatsa Mudigere
  28    Last edited 12/6/18.
  29    Inputs:
  30        points (nparray): two-dimensional array with each point of form [x f g]
  31        x_min_bound (float): minimum value that brackets minimum (default: minimum of points)
  32        x_max_bound (float): maximum value that brackets minimum (default: maximum of points)
  33        plot (bool): plot interpolating polynomial
  34    Outputs:
  35        x_sol (float): minimizer of interpolating polynomial
  36        F_min (float): minimum of interpolating polynomial
  37    Note:
  38      . Set f or g to np.nan if they are unknown
  39    """
  40    no_points = points.shape[0]
  41    order = np.sum(1 - np.isnan(points[:, 1:3]).astype('int')) - 1
  42
  43    x_min = np.min(points[:, 0])
  44    x_max = np.max(points[:, 0])
  45
  46    # compute bounds of interpolation area
  47    if x_min_bound is None:
  48        x_min_bound = x_min
  49    if x_max_bound is None:
  50        x_max_bound = x_max
  51
  52    # explicit formula for quadratic interpolation
  53    if no_points == 2 and order == 2 and plot is False:
  54        # Solution to quadratic interpolation is given by:
  55        # a = -(f1 - f2 - g1(x1 - x2))/(x1 - x2)^2
  56        # x_min = x1 - g1/(2a)
  57        # if x1 = 0, then is given by:
  58        # x_min = - (g1*x2^2)/(2(f2 - f1 - g1*x2))
  59
  60        if points[0, 0] == 0:
  61            x_sol = -points[0, 2] * points[1, 0] ** 2 / (2 * (points[1, 1] - points[0, 1] - points[0, 2] * points[1, 0]))
  62        else:
  63            a = -(points[0, 1] - points[1, 1] - points[0, 2] * (points[0, 0] - points[1, 0])) / (points[0, 0] - points[1, 0]) ** 2
  64            x_sol = points[0, 0] - points[0, 2]/(2*a)
  65
  66        x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
  67
  68    # explicit formula for cubic interpolation
  69    elif no_points == 2 and order == 3 and plot is False:
  70        # Solution to cubic interpolation is given by:
  71        # d1 = g1 + g2 - 3((f1 - f2)/(x1 - x2))
  72        # d2 = sqrt(d1^2 - g1*g2)
  73        # x_min = x2 - (x2 - x1)*((g2 + d2 - d1)/(g2 - g1 + 2*d2))
  74        d1 = points[0, 2] + points[1, 2] - 3 * ((points[0, 1] - points[1, 1]) / (points[0, 0] - points[1, 0]))
  75        d2 = np.sqrt(d1 ** 2 - points[0, 2] * points[1, 2])
  76        if np.isreal(d2):
  77            x_sol = points[1, 0] - (points[1, 0] - points[0, 0]) * ((points[1, 2] + d2 - d1) / (points[1, 2] - points[0, 2] + 2 * d2))
  78            x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
  79        else:
  80            x_sol = (x_max_bound + x_min_bound)/2
  81
  82    # solve linear system
  83    else:
  84        # define linear constraints
  85        A = np.zeros((0, order + 1))
  86        b = np.zeros((0, 1))
  87
  88        # add linear constraints on function values
  89        for i in range(no_points):
  90            if not np.isnan(points[i, 1]):
  91                constraint = np.zeros((1, order + 1))
  92                for j in range(order, -1, -1):
  93                    constraint[0, order - j] = points[i, 0] ** j
  94                A = np.append(A, constraint, 0)
  95                b = np.append(b, points[i, 1])
  96
  97        # add linear constraints on gradient values
  98        for i in range(no_points):
  99            if not np.isnan(points[i, 2]):
 100                constraint = np.zeros((1, order + 1))
 101                for j in range(order):
 102                    constraint[0, j] = (order - j) * points[i, 0] ** (order - j - 1)
 103                A = np.append(A, constraint, 0)
 104                b = np.append(b, points[i, 2])
 105
 106        # check if system is solvable
 107        if A.shape[0] != A.shape[1] or np.linalg.matrix_rank(A) != A.shape[0]:
 108            x_sol = (x_min_bound + x_max_bound)/2
 109            f_min = np.Inf
 110        else:
 111            # solve linear system for interpolating polynomial
 112            coeff = np.linalg.solve(A, b)
 113
 114            # compute critical points
 115            dcoeff = np.zeros(order)
 116            for i in range(len(coeff) - 1):
 117                dcoeff[i] = coeff[i] * (order - i)
 118
 119            crit_pts = np.array([x_min_bound, x_max_bound])
 120            crit_pts = np.append(crit_pts, points[:, 0])
 121
 122            if not np.isinf(dcoeff).any():
 123                roots = np.roots(dcoeff)
 124                crit_pts = np.append(crit_pts, roots)
 125
 126            # test critical points
 127            f_min = np.Inf
 128            x_sol = (x_min_bound + x_max_bound) / 2 # defaults to bisection
 129            for crit_pt in crit_pts:
 130                if np.isreal(crit_pt) and crit_pt >= x_min_bound and crit_pt <= x_max_bound:
 131                    F_cp = np.polyval(coeff, crit_pt)
 132                    if np.isreal(F_cp) and F_cp < f_min:
 133                        x_sol = np.real(crit_pt)
 134                        f_min = np.real(F_cp)
 135
 136            if(plot):
 137                plt.figure()
 138                x = np.arange(x_min_bound, x_max_bound, (x_max_bound - x_min_bound)/10000)
 139                f = np.polyval(coeff, x)
 140                plt.plot(x, f)
 141                plt.plot(x_sol, f_min, 'x')
 142
 143    return x_sol

 146class LBFGS(Optimizer):
 147    """
 148    Implements the L-BFGS algorithm. Compatible with multi-batch and full-overlap
 149    L-BFGS implementations and (stochastic) Powell damping. Partly based on the 
 150    original L-BFGS implementation in PyTorch, Mark Schmidt's minFunc MATLAB code, 
 151    and Michael Overton's weak Wolfe line search MATLAB code.
 152    Implemented by: Hao-Jun Michael Shi and Dheevatsa Mudigere
 153    Last edited 10/20/20.
 154    Warnings:
 155      . Does not support per-parameter options and parameter groups.
 156      . All parameters have to be on a single device.
 157    Inputs:
 158        lr (float): steplength or learning rate (default: 1)
 159        history_size (int): update history size (default: 10)
 160        line_search (str): designates line search to use (default: 'Wolfe')
 161            Options:
 162                'None': uses steplength designated in algorithm
 163                'Armijo': uses Armijo backtracking line search
 164                'Wolfe': uses Armijo-Wolfe bracketing line search
 165        dtype: data type (default: torch.float)
 166        debug (bool): debugging mode
 167    References:
 168    [1] Berahas, Albert S., Jorge Nocedal, and Martin Takác. "A Multi-Batch L-BFGS 
 169        Method for Machine Learning." Advances in Neural Information Processing 
 170        Systems. 2016.
 171    [2] Bollapragada, Raghu, et al. "A Progressive Batching L-BFGS Method for Machine 
 172        Learning." International Conference on Machine Learning. 2018.
 173    [3] Lewis, Adrian S., and Michael L. Overton. "Nonsmooth Optimization via Quasi-Newton
 174        Methods." Mathematical Programming 141.1-2 (2013): 135-163.
 175    [4] Liu, Dong C., and Jorge Nocedal. "On the Limited Memory BFGS Method for 
 176        Large Scale Optimization." Mathematical Programming 45.1-3 (1989): 503-528.
 177    [5] Nocedal, Jorge. "Updating Quasi-Newton Matrices With Limited Storage." 
 178        Mathematics of Computation 35.151 (1980): 773-782.
 179    [6] Nocedal, Jorge, and Stephen J. Wright. "Numerical Optimization." Springer New York,
 180        2006.
 181    [7] Schmidt, Mark. "minFunc: Unconstrained Differentiable Multivariate Optimization 
 182        in Matlab." Software available at http://www.cs.ubc.ca/~schmidtm/Software/minFunc.html 
 183        (2005).
 184    [8] Schraudolph, Nicol N., Jin Yu, and Simon Günter. "A Stochastic Quasi-Newton 
 185        Method for Online Convex Optimization." Artificial Intelligence and Statistics. 
 186        2007.
 187    [9] Wang, Xiao, et al. "Stochastic Quasi-Newton Methods for Nonconvex Stochastic 
 188        Optimization." SIAM Journal on Optimization 27.2 (2017): 927-956.

 227    def _gather_flat_grad(self):
 228        views = []
 229        for p in self._params:
 230            if p.grad is None:
 231                view = p.data.new(p.data.numel()).zero_()
 232            elif p.grad.data.is_sparse:
 233                view = p.grad.data.to_dense().view(-1)
 234            else:
 235                view = p.grad.data.view(-1)
 236            views.append(view)

 240        offset = 0

 243            # view as to avoid deprecated pointwise semantics
 244            p.data.add_(step_size, update[offset:offset + numel].view_as(p.data))
 245            offset += numel
 246        assert offset == self._numel()

 278    def two_loop_recursion(self, vec):
 279        """
 280        Performs two-loop recursion on given vector to obtain Hv.
 281        Inputs:
 282            vec (tensor): 1-D tensor to apply two-loop recursion to
 283        Output:
 284            r (tensor): matrix-vector product Hv
 285        """
 286
 287        group = self.param_groups[0]
 288        history_size = group['history_size']
 289
 290        state = self.state['global_state']
 291        old_dirs = state.get('old_dirs')    # change in gradients
 292        old_stps = state.get('old_stps')    # change in iterates
 293        H_diag = state.get('H_diag')
 294
 295        # compute the product of the inverse Hessian approximation and the gradient
 296        num_old = len(old_dirs)
 297
 298        if 'rho' not in state:
 299            state['rho'] = [None] * history_size
 300            state['alpha'] = [None] * history_size
 301        rho = state['rho']
 302        alpha = state['alpha']
 303
 304        for i in range(num_old):
 305            rho[i] = 1. / old_stps[i].dot(old_dirs[i])
 306
 307        q = vec
 308        for i in range(num_old - 1, -1, -1):
 309            alpha[i] = old_dirs[i].dot(q) * rho[i]
 310            q.add_(-alpha[i], old_stps[i])
 311
 312        # multiply by initial Hessian 
 313        # r/d is the final direction
 314        r = torch.mul(q, H_diag)
 315        for i in range(num_old):
 316            beta = old_stps[i].dot(r) * rho[i]
 317            r.add_(alpha[i] - beta, old_dirs[i])
 318
 319        return r

 321    def curvature_update(self, flat_grad, eps=1e-2, damping=False):
 322        """
 323        Performs curvature update.
 324        Inputs:
 325            flat_grad (tensor): 1-D tensor of flattened gradient for computing 
 326                gradient difference with previously stored gradient
 327            eps (float): constant for curvature pair rejection or damping (default: 1e-2)
 328            damping (bool): flag for using Powell damping (default: False)
 329        """
 330
 331        assert len(self.param_groups) == 1
 332
 333        # load parameters
 334        if(eps <= 0):
 335            raise(ValueError('Invalid eps; must be positive.'))
 336
 337        group = self.param_groups[0]
 338        history_size = group['history_size']
 339        debug = group['debug']
 340
 341        # variables cached in state (for tracing)
 342        state = self.state['global_state']
 343        fail = state.get('fail')
 344        
 345        # check if line search failed
 346        if not fail:
 347            
 348            d = state.get('d')
 349            t = state.get('t')
 350            old_dirs = state.get('old_dirs')
 351            old_stps = state.get('old_stps')
 352            H_diag = state.get('H_diag')
 353            prev_flat_grad = state.get('prev_flat_grad')
 354            Bs = state.get('Bs')
 355    
 356            # compute y's
 357            y = flat_grad.sub(prev_flat_grad)
 358            s = d.mul(t)
 359            sBs = s.dot(Bs)
 360            ys = y.dot(s)  # y*s
 361
 362            # update L-BFGS matrix
 363            if ys > eps * sBs or damping == True:
 364    
 365                # perform Powell damping
 366                if damping == True and ys < eps*sBs:
 367                    if debug:
 368                        print('Applying Powell damping...')
 369                    theta = ((1 - eps) * sBs)/(sBs - ys)
 370                    y = theta * y + (1 - theta) * Bs
 371    
 372                # updating memory
 373                if len(old_dirs) == history_size:
 374                    # shift history by one (limited-memory)
 375                    old_dirs.pop(0)
 376                    old_stps.pop(0)
 377    
 378                # store new direction/step
 379                old_dirs.append(s)
 380                old_stps.append(y)
 381    
 382                # update scale of initial Hessian approximation
 383                H_diag = ys / y.dot(y)  # (y*y)
 384                
 385                state['old_dirs'] = old_dirs
 386                state['old_stps'] = old_stps
 387                state['H_diag'] = H_diag
 388
 389            else:
 390                # save skip
 391                state['curv_skips'] += 1
 392                if debug:
 393                    print('Curvature pair skipped due to failed criterion')
 394
 395        else:
 396            # save skip
 397            state['fail_skips'] += 1
 398            if debug:
 399                print('Line search failed; curvature pair update skipped')
 400
 401        return

 403    def _step(self, p_k, g_Ok, g_Sk=None, options=None):
 404        """
 405        Performs a single optimization step.
 406        Inputs:
 407            p_k (tensor): 1-D tensor specifying search direction
 408            g_Ok (tensor): 1-D tensor of flattened gradient over overlap O_k used
 409                            for gradient differencing in curvature pair update
 410            g_Sk (tensor): 1-D tensor of flattened gradient over full sample S_k
 411                            used for curvature pair damping or rejection criterion,
 412                            if None, will use g_Ok (default: None)
 413            options (dict): contains options for performing line search (default: None)
 414        Options for Armijo backtracking line search:
 415            'closure' (callable): reevaluates model and returns function value
 416            'current_loss' (tensor): objective value at current iterate (default: F(x_k))
 417            'gtd' (tensor): inner product g_Ok'd in line search (default: g_Ok'd)
 418            'eta' (tensor): factor for decreasing steplength > 0 (default: 2)
 419            'c1' (tensor): sufficient decrease constant in (0, 1) (default: 1e-4)
 420            'max_ls' (int): maximum number of line search steps permitted (default: 10)
 421            'interpolate' (bool): flag for using interpolation (default: True)
 422            'inplace' (bool): flag for inplace operations (default: True)
 423            'ls_debug' (bool): debugging mode for line search
 424        Options for Wolfe line search:
 425            'closure' (callable): reevaluates model and returns function value
 426            'current_loss' (tensor): objective value at current iterate (default: F(x_k))
 427            'gtd' (tensor): inner product g_Ok'd in line search (default: g_Ok'd)
 428            'eta' (float): factor for extrapolation (default: 2)
 429            'c1' (float): sufficient decrease constant in (0, 1) (default: 1e-4)
 430            'c2' (float): curvature condition constant in (0, 1) (default: 0.9)
 431            'max_ls' (int): maximum number of line search steps permitted (default: 10)
 432            'interpolate' (bool): flag for using interpolation (default: True)
 433            'inplace' (bool): flag for inplace operations (default: True)
 434            'ls_debug' (bool): debugging mode for line search
 435        Outputs (depends on line search):
 436          . No line search:
 437                t (float): steplength
 438          . Armijo backtracking line search:
 439                F_new (tensor): loss function at new iterate
 440                t (tensor): final steplength
 441                ls_step (int): number of backtracks
 442                closure_eval (int): number of closure evaluations
 443                desc_dir (bool): descent direction flag
 444                    True: p_k is descent direction with respect to the line search
 445                    function
 446                    False: p_k is not a descent direction with respect to the line
 447                    search function
 448                fail (bool): failure flag
 449                    True: line search reached maximum number of iterations, failed
 450                    False: line search succeeded
 451          . Wolfe line search:
 452                F_new (tensor): loss function at new iterate
 453                g_new (tensor): gradient at new iterate
 454                t (float): final steplength
 455                ls_step (int): number of backtracks
 456                closure_eval (int): number of closure evaluations
 457                grad_eval (int): number of gradient evaluations
 458                desc_dir (bool): descent direction flag
 459                    True: p_k is descent direction with respect to the line search
 460                    function
 461                    False: p_k is not a descent direction with respect to the line
 462                    search function
 463                fail (bool): failure flag
 464                    True: line search reached maximum number of iterations, failed
 465                    False: line search succeeded
 466        Notes:
 467          . If encountering line search failure in the deterministic setting, one
 468            should try increasing the maximum number of line search steps max_ls.
 469        """
 470
 471        if options is None:
 472            options = {}
 473        assert len(self.param_groups) == 1
 474
 475        # load parameter options
 476        group = self.param_groups[0]
 477        lr = group['lr']
 478        line_search = group['line_search']
 479        dtype = group['dtype']
 480        debug = group['debug']
 481
 482        # variables cached in state (for tracing)
 483        state = self.state['global_state']
 484        d = state.get('d')
 485        t = state.get('t')
 486        prev_flat_grad = state.get('prev_flat_grad')
 487        Bs = state.get('Bs')
 488
 489        # keep track of nb of iterations
 490        state['n_iter'] += 1
 491
 492        # set search direction
 493        d = p_k
 494
 495        # modify previous gradient
 496        if prev_flat_grad is None:
 497            prev_flat_grad = g_Ok.clone()
 498        else:
 499            prev_flat_grad.copy_(g_Ok)
 500
 501        # set initial step size
 502        t = lr
 503
 504        # closure evaluation counter
 505        closure_eval = 0
 506
 507        if g_Sk is None:
 508            g_Sk = g_Ok.clone()
 509
 510        # perform Armijo backtracking line search
 511        if line_search == 'Armijo':
 512
 513            # load options
 514            if options:
 515                if 'closure' not in options.keys():
 516                    raise(ValueError('closure option not specified.'))
 517                else:
 518                    closure = options['closure']
 519
 520                if 'gtd' not in options.keys():
 521                    gtd = g_Sk.dot(d)
 522                else:
 523                    gtd = options['gtd']
 524
 525                if 'current_loss' not in options.keys():
 526                    F_k = closure()
 527                    closure_eval += 1
 528                else:
 529                    F_k = options['current_loss']
 530
 531                if 'eta' not in options.keys():
 532                    eta = 2
 533                elif options['eta'] <= 0:
 534                    raise(ValueError('Invalid eta; must be positive.'))
 535                else:
 536                    eta = options['eta']
 537
 538                if 'c1' not in options.keys():
 539                    c1 = 1e-4
 540                elif options['c1'] >= 1 or options['c1'] <= 0:
 541                    raise(ValueError('Invalid c1; must be strictly between 0 and 1.'))
 542                else:
 543                    c1 = options['c1']
 544
 545                if 'max_ls' not in options.keys():
 546                    max_ls = 10
 547                elif options['max_ls'] <= 0:
 548                    raise(ValueError('Invalid max_ls; must be positive.'))
 549                else:
 550                    max_ls = options['max_ls']
 551
 552                if 'interpolate' not in options.keys():
 553                    interpolate = True
 554                else:
 555                    interpolate = options['interpolate']
 556
 557                if 'inplace' not in options.keys():
 558                    inplace = True
 559                else:
 560                    inplace = options['inplace']
 561                    
 562                if 'ls_debug' not in options.keys():
 563                    ls_debug = False
 564                else:
 565                    ls_debug = options['ls_debug']
 566
 567            else:
 568                raise(ValueError('Options are not specified; need closure evaluating function.'))
 569
 570            # initialize values
 571            if interpolate:
 572                if torch.cuda.is_available():
 573                    F_prev = torch.tensor(np.nan, dtype=dtype).cuda()
 574                else:
 575                    F_prev = torch.tensor(np.nan, dtype=dtype)
 576
 577            ls_step = 0
 578            t_prev = 0 # old steplength
 579            fail = False # failure flag
 580
 581            # begin print for debug mode
 582            if ls_debug:
 583                print('==================================== Begin Armijo line search ===================================')
 584                print('F(x): %.8e  g*d: %.8e' % (F_k, gtd))
 585
 586            # check if search direction is descent direction
 587            if gtd >= 0:
 588                desc_dir = False
 589                if debug:
 590                    print('Not a descent direction!')
 591            else:
 592                desc_dir = True
 593
 594            # store values if not in-place
 595            if not inplace:
 596                current_params = self._copy_params()
 597
 598            # update and evaluate at new point
 599            self._add_update(t, d)
 600            F_new = closure()
 601            closure_eval += 1
 602
 603            # print info if debugging

280        """

282        Inputs:
283            vec (tensor): 1-D tensor to apply two-loop recursion to
284        Output:
285            r (tensor): matrix-vector product Hv

293        old_stps = state.get('old_stps')    # change in iterates
294        H_diag = state.get('H_diag')
295
296        # compute the product of the inverse Hessian approximation and the gradient
297        num_old = len(old_dirs)
298
299        if 'rho' not in state:
300            state['rho'] = [None] * history_size
301            state['alpha'] = [None] * history_size
302        rho = state['rho']
303        alpha = state['alpha']
304
305        for i in range(num_old):
306            rho[i] = 1. / old_stps[i].dot(old_dirs[i])
307
308        q = vec
309        for i in range(num_old - 1, -1, -1):
310            alpha[i] = old_dirs[i].dot(q) * rho[i]
311            q.add_(-alpha[i], old_stps[i])
312
313        # multiply by initial Hessian 
314        # r/d is the final direction
315        r = torch.mul(q, H_diag)
316        for i in range(num_old):
317            beta = old_stps[i].dot(r) * rho[i]
318            r.add_(alpha[i] - beta, old_dirs[i])

322    def curvature_update(self, flat_grad, eps=1e-2, damping=False):
323        """
324        Performs curvature update.
325        Inputs:
326            flat_grad (tensor): 1-D tensor of flattened gradient for computing 
327                gradient difference with previously stored gradient
328            eps (float): constant for curvature pair rejection or damping (default: 1e-2)
329            damping (bool): flag for using Powell damping (default: False)
330        """
331
332        assert len(self.param_groups) == 1
333
334        # load parameters
335        if(eps <= 0):
336            raise(ValueError('Invalid eps; must be positive.'))
337
338        group = self.param_groups[0]
339        history_size = group['history_size']
340        debug = group['debug']
341
342        # variables cached in state (for tracing)
343        state = self.state['global_state']
344        fail = state.get('fail')
345        
346        # check if line search failed
347        if not fail:
348            
349            d = state.get('d')
350            t = state.get('t')
351            old_dirs = state.get('old_dirs')
352            old_stps = state.get('old_stps')
353            H_diag = state.get('H_diag')
354            prev_flat_grad = state.get('prev_flat_grad')
355            Bs = state.get('Bs')
356    
357            # compute y's
358            y = flat_grad.sub(prev_flat_grad)
359            s = d.mul(t)
360            sBs = s.dot(Bs)
361            ys = y.dot(s)  # y*s
362
363            # update L-BFGS matrix
364            if ys > eps * sBs or damping == True:
365    
366                # perform Powell damping
367                if damping == True and ys < eps*sBs:
368                    if debug:
369                        print('Applying Powell damping...')
370                    theta = ((1 - eps) * sBs)/(sBs - ys)
371                    y = theta * y + (1 - theta) * Bs
372    
373                # updating memory
374                if len(old_dirs) == history_size:
375                    # shift history by one (limited-memory)
376                    old_dirs.pop(0)
377                    old_stps.pop(0)
378    
379                # store new direction/step
380                old_dirs.append(s)
381                old_stps.append(y)
382    
383                # update scale of initial Hessian approximation
384                H_diag = ys / y.dot(y)  # (y*y)
385                
386                state['old_dirs'] = old_dirs
387                state['old_stps'] = old_stps
388                state['H_diag'] = H_diag
389
390            else:
391                # save skip
392                state['curv_skips'] += 1
393                if debug:
394                    print('Curvature pair skipped due to failed criterion')

396        else:
397            # save skip
398            state['fail_skips'] += 1
