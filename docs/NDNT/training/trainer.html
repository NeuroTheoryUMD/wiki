109    def prepare_fit(self, model, seed=None):
110        '''
111        This is called before fit_loop, and is used to set up the model and optimizer.
112        '''
113        GPU_FLAG = torch.cuda.is_available()
114        GPU_USED = self.device.type == 'cuda'
115        if self.verbose > 0:
116            print("\nGPU Available: %r, GPU Used: %r" %(GPU_FLAG, GPU_USED))
117
118        if GPU_FLAG and GPU_USED:
119            self.use_gpu = True
120            torch.cuda.empty_cache()
121        else:
122            self.use_gpu = False
123
124        self.use_closure = isinstance(self.optimizer, torch.optim.LBFGS) or isinstance(self.optimizer, LBFGSNew)
125
126        # main training loop
127        if self.optimize_graph:
128            torch.backends.cudnn.benchmark = True # uses the inbuilt cudnn auto-tuner to find the fastest convolution algorithms.
129            # Note: On Nvidia GPUs you can add the following line at the beginning of our code.
130            # This will allow the cuda backend to optimize your graph during its first execution.
131            # However, be aware that if you change the network input/output tensor size the graph
132            # will be optimized each time a change occurs. This can lead to very slow runtime and out of memory errors.
133            # Only set this flag if your input and output have always the same shape.
134            # Usually, this results in an improvement of about 20%.
135        
136        if seed is not None:
137            # set flags / seeds    
138            np.random.seed(seed)
139            torch.manual_seed(seed)
140            torch.cuda.manual_seed(seed)
141        
142        model = model.to(self.device) # move model to device
143
144        self.optimizer.zero_grad() # set gradients to zero
145
146        return model

149    def fit_loop(self, model, epochs, train_loader, val_loader):
150        
151        self.nbatch = len(train_loader)
152
153        if self.fullbatch:
154            self.accumulate_grad_batches = len(train_loader)
155        self.accumulate_grad_batches = min([self.nbatch, self.accumulate_grad_batches])
156        
157        if self.log_activations:
158            activations = {}
159            self.hooks = []
160            self.logged_parameters = []
161
162            def hook(name):
163                def hook_fn(m, i, o):
164                    activations[name] = o
165
166                return hook_fn
167
168            # Hook the activations
169            for name, layer in model.named_modules():
170                if 'NL' in name or 'reg' in name or 'loss' in name:
171                    continue
172
173                self.logged_parameters.append(name)
174                self.hooks.append(layer.register_forward_hook(hook(name)))
175
176        # main loop for training
177        for epoch in range(epochs):
178            self.epoch += 1
179            # train one epoch
180            out = self.train_one_epoch(model, train_loader, self.epoch)
181            self.logger.add_scalar('Loss/Train (Epoch)', out['train_loss'], self.epoch)
182            train_loss = out['train_loss']
183            #if self.verbose > 1:
184            #    print('Train loss: %0.8f'%out['train_loss'])
185            if np.isnan(train_loss):
186                self.graceful_exit(model)
187
188            if self.log_activations:
189                for name in self.logged_parameters:
190                    try:
191                        self.logger.add_histogram(
192                                    f"Activations/{name.replace('.', ' ')}/hist",
193                                    activations[name].view(-1),
194                                    self.epoch,
195                                )
196
197                        self.logger.add_scalar(
198                                    f"Activations/{name.replace('.', ' ')}/mean",
199                                    activations[name].mean(),
200                                    self.epoch,
201                                )
202
203                        self.logger.add_scalar(
204                                    f"Activations/{name.replace('.', ' ')}/std",
205                                    activations[name]
206                                    .std(dim=0)
207                                    .mean(),
208                                    self.epoch,
209                                )
210                    except:
211                        pass
212
213            # validate every epoch
214            if self.epoch % 1 == 0:
215                out = self.validate_one_epoch(model, val_loader)
216                if out['val_loss'] < self.val_loss_min:
217                    is_best=True
218                else:
219                    is_best=False
220                self.val_loss_min = out['val_loss']
221                self.logger.add_scalar('Loss/Validation (Epoch)', self.val_loss_min, self.epoch)
222            
223            if self.verbose==1:
224                print("Epoch %d: train loss %.6f val loss %.6f" %(self.epoch, train_loss, out['val_loss']))
225                
226            # scheduler if scheduler steps at epoch level
227            if self.scheduler:
228                if self.step_scheduler_after == "epoch":
229                    if self.step_scheduler_metric is None:
230                        self.scheduler.step()
231                    else:
232                        step_metric = out[self.step_scheduler_metric]
233                        self.scheduler.step(step_metric)
234            
235            # checkpoint
236            self.checkpoint_model(model, self.epoch, is_best=is_best)
237
238            # callbacks: e.g., early stopping
239            if self.early_stopping:
240                self.early_stopping(out['val_loss'])
241                if self.early_stopping.early_stop:
242                    if self.verbose > 0:
243                        print("Early stopping")
244                    break

312        
313        # Data to device if it's not already there

315            for dsub in data:
