142    def forward(self, x):
143        """
144        Forward pass through the IterLayer.
145
146        Args:
147            x: torch.Tensor, input tensor
148
149        Returns:
150            y: torch.Tensor, output tensor
151        """
152
153        # Reshape weight matrix and inputs (note uses 4-d rep of tensor before combinine dims 0,3)
154        #return x + super().forward(x)
155        ## THIS IS CUT-AND-PASTE FROM ConvLayer -- with mods
156
157        # Prepare stimulus
158        x = x.reshape([-1]+self.input_dims).permute(0,1,4,2,3)
159        if self.is1D:
160            x = torch.reshape( x, (-1, self.folded_dims, self.input_dims[1]) )
161        else:
162            x = torch.reshape( x, (-1, self.folded_dims, self.input_dims[1], self.input_dims[2]) )
163
164        # Prepare weights
165        w = self.preprocess_weights().reshape(self.filter_dims+[self.num_filters]).permute(4,0,3,1,2)
166        # puts output_dims first, as required for conv
167
168        outputs = None
169        for iter in range(self.num_iter):
170
171            if self.LN_reverse:
172                if self.NL is not None:
173                    x = self.NL(x)
174
175            if self.is1D:
176
177                if self._fullpadding:
178                    #s = F.pad(s, self._npads, "constant", 0)
179                    y = F.conv1d(
180                        F.pad(x, self._npads, "constant", 0),
181                        w.view([-1, self.folded_dims, self.filter_dims[1]]), 
182                        bias=self.bias,
183                        stride=self.stride, dilation=self.dilation)
184                else:
185                    y = F.conv1d(
186                        x,
187                        w.view([-1, self.folded_dims, self.filter_dims[1]]), 
188                        bias=self.bias,
189                        padding=self._npads[0],
190                        stride=self.stride, dilation=self.dilation)
191
192            else:
193
194                if self._fullpadding:
195                    x = F.pad(x, self._npads, "constant", 0)
196                    y = F.conv2d(
197                        x, # we do our own padding
198                        w.view([-1, self.folded_dims, self.filter_dims[1], self.filter_dims[2]]),
199                        bias=self.bias,
200                        stride=self.stride,
201                        dilation=self.dilation)
202                else:
203                    # functional pads since padding is simple
204                    y = F.conv2d(
205                        x, 
206                        w.reshape([-1, self.folded_dims, self.filter_dims[1], self.filter_dims[2]]),
207                        padding=(self._npads[2], self._npads[0]),
208                        bias=self.bias,
209                        stride=self.stride,
210                        dilation=self.dilation)
211
212
213            # Nonlinearity
214            if (self.NL is not None) &  (not self.LN_reverse):
215                y = self.NL(y)
216
217            if self.res_layer:
218                y = y + x  # Add input back in
219
220            if self._ei_mask is not None:
221                if self.is1D:
222                    y = y * self._ei_mask[None, :, None]
223                else:
224                    y = y * self._ei_mask[None, :, None, None]
225
226            if self.output_norm is not None:
227                y = self.output_norm[iter](y)
228
229            if (self.output_config == 'full') | (iter == self.num_iter-1):
230                if outputs is None:
231                    outputs = y
232                else:
233                    outputs = torch.cat( (outputs, y), dim=1)
234            x = y
235        # end of iteration loop
236
237        return torch.reshape(outputs, (-1, self.num_outputs))
