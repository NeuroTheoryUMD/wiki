151    def __init__(self, name, desc, objective_func, num_trials=0, trial_paramses=None, models_dir='../models', overwrite=False):
152        """
153        Class to run a study.
154
155        Args:
156            name (str): the name of the study
157            desc (str): the description of the study
158            objective_func (function): the function to optimize
159            num_trials (int): the number of trials to run
160            trial_paramses (list): the list of trial parameters to run
161            models_dir (str): the directory to save the models
162            overwrite (bool): whether to overwrite the models directory
163
164        Raises:
165            ValueError: if the model directory already exists
166        """
167        assert num_trials > 0 or trial_paramses is not None, 'Either num_trials or trial_paramses must be specified!'
168        
169        self.name = name
170        self.desc = desc
171        self.objective_func = objective_func
172        self.trial_paramses = trial_paramses
173        self.num_trials = num_trials
174        self.models_dir = models_dir
175        self.overwrite = overwrite
176        self.study_dir = os.path.join(models_dir, name)
177        self.study_path = os.path.join(models_dir, name, 'study.pkl')
178        self.study = optuna.create_study(
179                                direction='maximize',
180                                study_name=desc)
181
182        # to track current trial info
183        self.current_study = None
184        self.current_trial = None
185        
186        # add the number of queued trials to the total number of trials
187        if trial_paramses is not None:
188            self.num_trials = len(trial_paramses) + self.num_trials
189
190        # make the models_dir if it doesn't exist, 
191        # or throw an error
192        if not os.path.exists(self.study_dir):
193            os.makedirs(self.study_dir)
194        else:
195            if overwrite:
196                print('Overwriting model directory!')
197            else:
198                raise ValueError('Model directory already exists!')
199        
200        # load the previous study trials
201        # and use them if they are finished
202        # and delete unfinished pickles
203        if os.path.exists(self.study_path):
204            with open(self.study_path, 'rb') as f:
205                prev_study = pickle.load(f)
206            for trial in prev_study.trials:
207                if trial.state == optuna.trial.TrialState.COMPLETE:
208                    self.study.add_trial(trial)
209                else:
210                    # remove the previous pickle (if it exists)
211                    if os.path.exists(os.path.join(self.study_dir, 'trial_{}.pkl'.format(trial.number))):
212                        os.remove(os.path.join(self.study_dir, 'trial_{}.pkl'.format(trial.number)))
213                    # remove the previous checkpoints (if they exist)
214                    if os.path.exists(os.path.join(self.study_dir, 'checkpoints', 'trial_{}'.format(trial.number))):
215                        os.rmdir(os.path.join(self.study_dir, 'checkpoints', 'trial_{}'.format(trial.number)))
216        print('Added', len(self.study.trials), 'trials from the previous studies.')
217
218        # enqueue the trial parameters
219        if self.trial_paramses is not None:
220            # validate all trial_params have the same keys
221            keys = set(self.trial_paramses[0].keys())
222            for trial_params in self.trial_paramses:
223                assert keys == set(trial_params.keys()), 'All trial_params must have the same keys!'
224
225            for trial_params in self.trial_paramses:
226                self.study.enqueue_trial(trial_params)
227
228        # TODO: make a JSON file to store the model versions and checkpoints directory
229
230    def save(self, trial):
231        """
232        Save the current trial and study.
233
234        Args:
235            trial (optuna.trial): the current trial
236        """
237        self.current_trial = trial
238
239    def post_objective_func(self, study, frozen_trial):
240        """
241        Save the trial and study after the objective function is run.
242
243        Args:
244            study (optuna.study): the current study
245            frozen_trial (optuna.trial): the frozen trial
246        """
247        # save the trial
248        print('Saving trial {}'.format(frozen_trial.number))
249        self.current_trial.trial = frozen_trial
250        name = self.current_trial.name
251        with open(os.path.join(self.study_dir, name+'.pkl'), 'wb') as f:
252            pickle.dump(self.current_trial, f)
253        self.current_trial = None
254        # save the study
255        with open(self.study_path, 'wb') as f:
256            pickle.dump(study, f)
257        # update the current study with the incremental study
258        self.current_study = study
259
260    def run(self):
261        """
262        Run the study.
263
264        Returns:
265            int: the best trial number
266            dict: the best trial parameters
267        """
268        try:
269            # dump the study at the beginning
270            with open(self.study_path, 'wb') as f:
271                pickle.dump(self.study, f)
272
273            self.study.optimize(lambda trial: self.objective_func(trial, 
274                                                                  'trial_{}'.format(trial.number), 
275                                                                  os.path.join(self.study_dir, 'checkpoints'),
276                                                                  self.save),
277                                callbacks=[self.post_objective_func],
278                                n_trials=self.num_trials, n_jobs=1)
279            
280            # print the best trial
281            return self.study.best_trial.number, self.study.best_params
282        except KeyboardInterrupt:
283            print('KeyboardInterrupt: exiting')
284            # TODO: keep track of the current checkpoint to resume the current trial and study
285            return self.study.best_trial.number, self.study.best_params

288    def __init__(self, study_name, models_dir='../models'):
289        """
290        Class to load a study.
291
292        Args:
293            study_name (str): the name of the study
294            models_dir (str): the directory where the models are stored
295        """
296        self.study_name = study_name
297        self.study_dir = os.path.join(models_dir, study_name)
298        self.study_path = os.path.join(models_dir, study_name, 'study.pkl')
299        self.study = None
300        self.trials = []
301
302    def load(self, load_data=False, build_stim=False):
303        """
304        Load the study.
305
306        Args:
307            load_data (bool): whether to load the data
308            build_stim (bool): whether to build the stimulus
309
310        Returns:
311            None
312        """
313        self.trials = []
314        # load the trials and the study from the models_dir
315        # load the study
316        with open(self.study_path, 'rb') as f:
317            self.study = pickle.load(f)
318        # load the trials referenced in the study
319        for trial in self.study.trials:
320            # skip trials that are not complete or waiting (for completion)
321            if trial.state != optuna.trial.TrialState.COMPLETE:
322                continue
323            model = Trial.load(self.study_name, 'trial_{}'.format(trial.number))
324            self.trials.append(model)
325
326        # load the data
327        if load_data:
328            self.load_data(build_stim)
329
330    def load_data(self, build_stim=False):
331        """
332        Load the data for the study.
333
334        Args:
335            build_stim (bool): whether to build the stimulus
336
337        Returns:
338            None
339        """
340        # determine the unique expts in all of the trials
341        expts = []
342        trial_expts = {}
343        for trial in self.trials:
344            if trial.expts not in expts:
345                expts.append(trial.expts)
346            trial_expts[trial] = expts.index(trial.expts)
347        print('Loading {} datasets for {} trials.'.format(len(expts), len(self.trials)))
348        # load the data for each expt
349        for expt in expts:
350            expt.load(build_stim=build_stim)
351        # assign the data to each trial
352        for trial in self.trials:
353            trial.expts = expts[trial_expts[trial]]
