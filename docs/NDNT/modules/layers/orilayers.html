 46        """
 47        Create a rotation matrix tensor for each angle in theta_list.
 48
 49        Args:
 50            filter_dims: filter dimensions
 51            theta_list: list of angles in degrees
 52
 53        Returns:
 54            rotation_matrix_tensor: rotation matrix tensor
 55        """
 56        assert self.filter_dims[2] != 1, "OriLayer: Stimulus must be 2-D"
 57
 58        w = filter_dims[1]
 59        if w%2 == 1:             
 60            x_pos = np.repeat(np.linspace(-np.floor(w/2), np.floor(w/2), w), w)
 61            y_pos = np.tile(np.linspace(-np.floor(w/2), np.floor(w/2), w), w)
 62        else:
 63            x_pos = np.repeat((np.linspace(-(w-1), w-1, w)), w)
 64            y_pos = np.tile((np.linspace(-(w-1), w-1, w)), w)
 65
 66        indices=[]
 67        for k in range(len(theta_list)):
 68            assert theta_list[k]==int(theta_list[k]), "OriLayer: All angles must be in degrees!"
 69
 70            theta=theta_list[k]*np.pi/180
 71            rotation_matrix=rotation_matrix=np.array(
 72                [np.cos(theta), -np.sin(theta), np.sin(theta), np.cos(theta)]).reshape(2, 2)
 73            
 74            for i in range(len(x_pos)):
 75                vector = np.array([x_pos[i], y_pos[i]]).reshape(2, 1)
 76                rotated_vector = np.matmul(rotation_matrix, vector)     
 77                integer_center = np.round(rotated_vector)
 78
 79                if w%2 == 0: 
 80                    frac_part,_ =np.modf(integer_center)
 81                    for j in range(2):
 82                        if integer_center[j]%2==0:
 83                            if abs(frac_part[j])<0.5:
 84                                integer_center[j]+=np.sign(integer_center[j])
 85                            else:
 86                                integer_center[j]-=np.sign(integer_center[j])
 87
 88                x_loc = np.argwhere(x_pos == integer_center[0])
 89                if len(x_loc) == w:
 90                    x_search = x_loc.reshape(w)
 91                    y_loc = x_search[np.argwhere(y_pos[x_search] == integer_center[1])]
 92
 93                    if len(y_loc) == 0:
 94                        continue
 95                    else:
 96                        indices.append([k, i, int(y_loc[0, 0])])
 97                else:
 98                    continue
 99        
100        rotation_matrix_tensor=torch.sparse_coo_tensor(
101            torch.tensor(indices).t(), torch.ones(len(indices)), size=(len(theta_list), w**2, w**2))
102        
103        return rotation_matrix_tensor

105    def forward(self, x):
106        """
107        Forward pass through the layer.
108
109        Args:
110            x: torch.Tensor, input tensor of shape (batch_size, *input_dims)
111
112        Returns:
113            y: torch.Tensor, output tensor of shape (batch_size, *output_dims)
114        """
115        w = self.preprocess_weights() #What shape is this? (whatever self.shape is)
116        #You still need the original! w is NC*NXY*NT by NF 
117        x_0 = torch.matmul(x, w) #linearity 
118        if self.norm_type == 2:
119            x_0 = x_0 / self.weight_scale #normalization 
120
121        x_0 = x_0 + self.bias #bias 
122
123        #if self.output_norm is not None:
124        #    x = self.output_norm(x)
125
126        # Nonlinearity
127        if self.NL is not None:
128            x_0 = self.NL(x_0)
129
130        # Constrain output to be signed
131        if self._ei_mask is not None:
132            x_0 = x_0 * self._ei_mask
133        
134        x_hats = torch.zeros(tuple([x_0.shape[0], x_0.shape[1], len(self.angles)+1])) 
135        x_hats[:,:,0] = x_0 # Wrangle shapes now. 
136        w_slicing = w.reshape(self.filter_dims[0], self.filter_dims[1], self.filter_dims[2], self.filter_dims[3], self.num_filters)
137        w_flattened = w_slicing.reshape(self.filter_dims[1]*self.filter_dims[2], self.filter_dims[0]*self.filter_dims[3]*self.num_filters) 
138
139        for i in range(len(self.angles)):
140            w_theta = torch.sparse.mm(self.rotation_matrices[i], w_flattened)
141            w_reshaped = w_theta.reshape(self.shape) #NC*NXY*NT by NF 
142            x_theta = torch.sparse.mm(x, w_reshaped) #Dimensionality! (x is B by NC*NXY*NT)
143            if self.norm_type == 2:
144                x_theta = x_theta / self.weight_scale
145            x_theta = x_theta + self.bias
146            if self.NL is not None:
147                x_theta = self.NL(x_theta)
148            if self._ei_mask is not None:
149                x_theta = x_theta * self._ei_mask
150            #self.activity_regularization = self.activity_reg.regularize(x_theta) #Do we need to change that? 
151            x_hats[:,:,i+1] = x_theta
152
153        # store activity regularization to add to loss later
154        if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model
155            self.reg.compute_activity_regularization(x_hats)
156
157        return x_hats.reshape([x_0.shape[0], -1])

258        """

286

288                vector = np.array([x_pos[i], y_pos[i]]).reshape(2, 1)
289                rotated_vector = np.matmul(rotation_matrix, vector)

291                if w%2 == 0:

348    def forward(self, x):
349        """
350        Forward pass through the layer.
351
352        Args:
353            x: torch.Tensor, input tensor of shape (batch_size, *input_dims)
354
355        Returns:
356            y: torch.Tensor, output tensor of shape (batch_size, *output_dims)
357        """
358        if self._padding == 'circular':
359            pad_type = 'circular'
360        else:
361            pad_type = 'constant'
362        
363        s = x.reshape(-1, self.input_dims[0], self.input_dims[1], self.input_dims[2])
364
365        w = self.preprocess_weights()
366        # permute num_filters, input_dims[0], width, height, (no lag)
367        w = w.reshape(self.filter_dims[:3]+[self.num_filters]).permute(3, 0, 1, 2)
368        # combine num_filters*input_dims[0] and width*height into one dimension
369
370        #w_flattened = w.reshape(self.num_filters*self.filter_dims[0], self.filter_dims[1]*self.filter_dims[2]) 
371        w_prep = w.reshape(self.num_filters*self.filter_dims[0], self.filter_dims[1], self.filter_dims[2])
372
373        # repeat weights for each angle along a new dimension
374        # 1, 1 specifies to keep the filters and width*height dimensions the same
375        #rotated_ws = w_flattened.repeat(len(self.angles), 1, 1).permute(1, 2, 0)
376        
377        #rotated_ws = w_flattened.repeat(len(self.angles), 1, 1).permute(1, 2, 0)
378        rotated_ws = w_prep.repeat(len(self.angles), 1, 1, 1).permute(1, 2, 3, 0)
379        
380        # use torch.sparse.mm to multiply the rotation matrices by the weights
381        for ii in range(1, len(self.angles)):  # first angle always = 0 (so explicit copy -- can skip)
382            # get the weights for the given angle
383            #w_theta = rotated_ws[:, :, i]
384
385            # sparse matmul method, but causes artifacting at non 90 degree angles
386            # rotate the weight matrix for the given angle
387            #w_theta = torch.sparse.mm(w_flattened, self.rotation_matrices[i])
388            
389            # rotate using torchvision transform
390            w_theta = TF.rotate(
391                img=w_prep,
392                interpolation=torchvision.transforms.InterpolationMode.BILINEAR,
393                angle=float(self.angles[ii]))
394            #    img=w_flattened.reshape(-1, self.filter_dims[1], self.filter_dims[2]),
395            #    interpolation=torchvision.transforms.InterpolationMode.BILINEAR,
396            #    angle=float(self.angles[ii])).reshape(-1, self.filter_dims[1]*self.filter_dims[2])
397
398            # put w_theta back into the full weight matrix
399            #rotated_ws[:, :, ii] = w_theta
400            rotated_ws[:, :, :, ii] = w_theta
401        
402        # reshape the weights so we can put the angles in the second dimension
403        rotated_ws_reshaped = rotated_ws.reshape((self.num_filters,
404                                                  self.filter_dims[0], # in filters
405                                                  self.filter_dims[1], # width
406                                                  self.filter_dims[2], # height
407                                                  len(self.angles)))
408
409        # move the angles to the second dimension
410        rotated_ws_reshaped = rotated_ws_reshaped.permute(0, 4, 1, 2, 3)
411        # and combine the filters and angles into the folded_dims dimension
412        # since we are convolving in the folded_dims dimension to do all filters at once
413        rotated_ws_reshaped = rotated_ws_reshaped.reshape((self.num_filters*len(self.angles),
414                                                           self.filter_dims[0],
415                                                           self.filter_dims[1],
416                                                           self.filter_dims[2]))
417
418        if self._fullpadding:
419            s_padded = F.pad(s, self.npads, pad_type, 0)
420            y = F.conv2d(s_padded,
421                         rotated_ws_reshaped, 
422                         bias=self.bias.repeat(len(self.angles)),
423                         stride=self.stride,
424                         dilation=self.dilation)
425        else:
426            if self.padding == 'circular':
427                s_padded = F.pad(s, self._npads, pad_type, 0)
428                y = F.conv2d(s_padded, rotated_ws_reshaped,
429                             bias=self.bias.repeat(len(self.angles)),
430                             stride=self.stride,
431                             dilation=self.dilation)
432            else: # this is faster if not circular
433                y = F.conv2d(s, rotated_ws_reshaped,
434                             padding=(self._npads[2], self._npads[0]),
435                             bias=self.bias.repeat(len(self.angles)),
436                             stride=self.stride,
437                             dilation=self.dilation)
438
439        if not self.res_layer:
440            if self.output_norm is not None:
441                y = self.output_norm(y)
442        
443        if self.NL is not None:
444            y = self.NL(y)
445        if self._ei_mask is not None:
446            y = y*self._ei_mask[None, :, None, None]
447
448        # if self.res_layer:
449        #     y = y+torch.reshape(s, (-1, self.folded_dims, self.input_dims[1], self.input_dims[2]) )
450        #     if self.output_norm is not None:
451        #         y = self.output_norm(y)
452        
453        # # store activity regularization to add to loss later
454        # if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model
455        #     self.reg.compute_activity_regularization(y)
456
457        # pull the filters and angles apart again
458        y = y.reshape(-1, # the batch dimension
459                      self.num_filters,
460                      len(self.angles),
461                      self.output_dims[1], 
462                      self.output_dims[2])
463        # reshape y to have the orientiations in the last column
464        y = y.permute(0, 1, 3, 4, 2)
465        # flatten the last dimensions
466        return y.reshape(-1, # the batch dimension
467                         self.num_outputs)
